Inline math: $e^{i\pi}+1=0$.

\begin{equation}

\end{equation}

\begin{align}
f(x) &= x^2 + 2x + 1 \\
     &= (x+1)^2
\end{align}

\begin{theorem}
For all real numbers $x$, we have $x^2 \ge 0$.
\end{theorem}

\begin{proof}
Square of a real number is non-negative.
\end{proof}

\begin{lemma}
If $x > 1$, then $x^2 > 1$.
\end{lemma}

\begin{proof}
Immediate from monotonicity on positive reals.
\end{proof}
